\section{Docker}
 
\defverbatim[colored]\lstinstall{
 \begin{lstlisting}[language=python,showstringspaces=false, basicstyle={\tiny}, keywordstyle=\color{red}]

$ sudo apt-get update

$ sudo apt-get install apt-transport-https ca-certificates curl 
  software-properties-common

$ curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo apt-key add -

$ sudo add-apt-repository "deb [arch=amd64] https://download.docker.com/
  linux/ubuntu $(lsb_release -cs) stable"

$ sudo apt-get update

$ sudo apt-get install docker-ce
 \end{lstlisting}
}


\defverbatim[colored]\lsthelloworld{
 \begin{lstlisting}[language=python,showstringspaces=false, basicstyle={\tiny}, keywordstyle=\color{red}]

$ nano package.json

$ sudo docker run -p 8080:8080 -v $PWD/1:/src -d adsoft/nodejs
$ sudo docker ps

 \end{lstlisting}
}
